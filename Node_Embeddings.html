# Fonction pour extraire la liste des artes du graphe
def graph_to_edge_list(G):
    """
    Convertit le graphe en une liste d'artes.

    Parameters:
        G (networkx.Graph): Le graphe.

    Returns:
        list: Liste des artes du graphe.
    """
    edge_list = []
    for edge in G.edges():
        edge_list.append(edge)
    return edge_list

# Fonction pour convertir la liste d'artes en tenseur
def edge_list_to_tensor(edge_list):
    """
    Convertit une liste d'artes en un tenseur.

    Parameters:
        edge_list (list): Liste d'artes.

    Returns:
        torch.Tensor: Tenseur reprsentant les artes.
    """
    edge_index = torch.tensor(edge_list, dtype=torch.long).t().contiguous()
    return edge_index

import random

# Fonction pour chantillonner des artes ngatives
def sample_negative_edges(G, num_neg_samples):
    """
    chantillonne des artes ngatives.

    Parameters:
        G (networkx.Graph): Le graphe.
        num_neg_samples (int): Le nombre d'artes ngatives  chantillonner.

    Returns:
        list: Liste d'artes ngatives chantillonnes.
    """
    neg_edge_list = []
    for _ in range(num_neg_samples):
        nodes = random.sample(G.nodes(), 2)
        while G.has_edge(nodes[0], nodes[1]):
            nodes = random.sample(G.nodes(), 2)
        neg_edge_list.append(tuple(nodes))
    return neg_edge_list

# Fonction pour crer les embeddings des nuds
def create_node_emb(num_node=34, embedding_dim=16):
    """
    Cre des embeddings pour les nuds du graphe.

    Parameters:
        num_node (int): Nombre de nuds dans le graphe.
        embedding_dim (int): Dimension des embeddings.

    Returns:
        torch.nn.Embedding: La couche d'embedding.
    """
    emb = nn.Embedding(num_embeddings=num_node, embedding_dim=embedding_dim)
    shape = emb.weight.data.shape
    emb.weight.data = torch.rand(shape)
    return emb

emb = create_node_emb()

# Fonction pour visualiser les embeddings
def visualize_emb(emb):
    """
    Visualise les embeddings des nuds en utilisant l'analyse en composantes principales (PCA).

    Parameters:
        emb (torch.nn.Embedding): La couche d'embedding.
        G (networkx.Graph): Le graphe.
    """
    X = emb.weight.data.numpy()
    pca = PCA(n_components=2)
    components = pca.fit_transform(X)
    plt.figure(figsize=(6, 6))
    club1_x, club1_y, club2_x, club2_y = [], [], [], []
    for node in G.nodes(data=True):
        if node[1]['club'] == 'Mr. Hi':
            club1_x.append(components[node[0]][0])
            club1_y.append(components[node[0]][1])
        else:
            club2_x.append(components[node[0]][0])
            club2_y.append(components[node[0]][1])
    plt.scatter(club1_x, club1_y, color="red", label="Mr. Hi")
    plt.scatter(club2_x, club2_y, color="blue", label="Officer")
    plt.legend()
    plt.show()

from torch.optim import SGD

# Fonction pour calculer l'exactitude
def accuracy(pred, label):
    """
    Calcule l'exactitude des prdictions.

    Parameters:
        pred (torch.Tensor): Les prdictions.
        label (torch.Tensor): Les vraies tiquettes.

    Returns:
        float: L'exactitude.
    """
    accu = torch.sum(torch.round(pred) == label) / pred.shape[0]
    return accu

# Fonction d'entranement du modle
def train(emb, loss_fn, sigmoid, train_label, train_edge):
    """
    Entrane le modle d'embedding de graphe.

    Parameters:
        emb (torch.nn.Embedding): La couche d'embedding.
        loss_fn (torch.nn.Module): La fonction de perte.
        sigmoid (torch.nn.Module): La fonction d'activation sigmoid.
        train_label (torch.Tensor): Les tiquettes d'entranement.
        train_edge (torch.Tensor): Les artes d'entranement.
    """
    epochs = 1000
    learning_rate = 0.05
    optimizer = SGD(emb.parameters(), lr=learning_rate, momentum=0.9)

    for i in range(epochs):
        embeddings = emb(train_edge)
        logits = torch.sum(embeddings[0] * embeddings[1], dim=1)
        preds = sigmoid(logits)
        loss = loss_fn(preds, train_label)
        accu = accuracy(preds, train_label)

        optimizer.zero_grad()
        loss.backward()
        optimizer.step()

        if i % 50 == 0:
            print(f'Epoch {i}, Loss: {loss.item()}, Accuracy: {accu.item()}')

# Entranement du modle
loss_fn = nn.BCEWithLogitsLoss()
sigmoid = nn.Sigmoid()
